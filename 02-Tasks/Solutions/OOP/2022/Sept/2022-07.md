Отговор:
```c++
#include <iostream>

template <typename T>
class Array {
	static const size_t size = 10;
	T data[size];
public:
	T& at(size_t index) {
		if (index >= size)
			throw std::out_of_range("error");
		return data[index];
	}
};

void test() {
	Array<int> a;
	a.at(0) = 5;
	std::cout << a.at(0);
	Array<Array<int>> b;
	b.at(0) = a;
	std::cout << b.at(0).at(0);
}
```
